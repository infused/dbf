def table_path(name)
  glob = File.join(@dirname, "#{name}.dbf")
  path = Dir.glob(glob, File::FNM_CASEFOLD).first

  raise DBF::FileNotFoundError, "related table not found: #{name}" unless path && File.exist?(path)

  path
end
