    def find(command, options = {})
      case command
      when Fixnum
        record(command)
      when :all
        return records if options.empty?
        records.select do |record|
          options.map {|key, value| record.attributes[key.to_s] == value}.all?
        end
      when :first
        return records.first if options.empty?
        records.detect do |record|
          options.map {|key, value| record.attributes[key.to_s] == value}.all?
        end
      end
    end
