def header_size
  case version
  when '02'
    DBASE2_HEADER_SIZE
  when '04', '8c'
    DBASE7_HEADER_SIZE
  else
    DBASE3_HEADER_SIZE
  end
end

def memo_class

def read_column_args(table, io)
  case version
  when '02' then [table, *io.read(header_size * 2).unpack('A11 a C'), 0]
  when '04', '8c' then [table, *io.read(48).unpack('A32 a C C x13')]
  else [table, *io.read(header_size).unpack('A11 a x4 C2')]
  end
end
